## Install

```bash
npm install @domi7891/vuejs-datetime-picker
```

```javascript
import { Datetimepicker, Datepicker, Timepicker } from 'vuejs-datetime-picker'

export default {
  // ...
  components: {
    Datetimepicker,
    Datepicker,
    Timepicker,
    //...
  },
  //...
}
```

## Usage

```html
<datetimepicker></datetimepicker>
<!-- or -->
<datepicker></datepicker>
<!-- or -->
<timepicker></timepicker>
```
